/**
 * Status Handler untuk SKP Tenaga Ahli
 *
 * @package     Asosiasi
 * @subpackage  Assets/JS/SKP_Tenaga_Ahli
 * @version     1.0.1
 * @author      arisciwek
 *
 * Description: Menangani semua interaksi terkait perubahan status
 *              dan riwayat status SKP Tenaga Ahli
 *
 * Changelog:
 * 1.0.1 - 2024-12-12
 * - Added detailed logging
 * - Fixed history loading
 * - Improved error handling
 */

var AsosiasiSKPTenagaAhliStatus = {};

(function($) {
    'use strict';

    // Private methods container
    const methods = {
        init: function() {
            console.log('AsosiasiSKPTenagaAhliStatus.init called');
            this.initStatusChangeHandlers();
            this.initModalHandlers();
        },

        initStatusChangeHandlers: function() {
            console.log('Initializing SKP Tenaga Ahli status handlers');
            
            // Toggle status select dropdown
            $('#skp-tenaga-ahli-section').on('click', '.status-change-trigger', function(e) {
                e.preventDefault();
                e.stopPropagation();
                const $wrapper = $(this).closest('.status-wrapper');
                const $select = $wrapper.find('.status-select');
                
                $('#skp-tenaga-ahli-section .status-select').not($select).hide();
                $select.toggle();
            });

            // Handle status selection
            $('#skp-tenaga-ahli-section').on('change', '.status-select select', function() {
                const $select = $(this);
                const skpId = $select.data('id');
                const oldStatus = $select.data('current');
                const newStatus = $select.val();

                if (!newStatus) return;

                console.log('Status change selected:', {skpId, oldStatus, newStatus});

                // Hide select wrapper
                $select.closest('.status-select').hide();

                // Populate modal fields
                $('#status_skp_id').val(skpId);
                $('#status_old_status').val(oldStatus);
                $('#status_new_status').val(newStatus);
                $('#status_reason').val('');
                $('#status-change-modal-tenaga-ahli').show();
                $('#status_reason').focus();
                
                // Reset select
                $select.val('');
            });

            // Close dropdowns when clicking outside
            $(window).on('click', function(e) {
                if (!$(e.target).closest('#skp-tenaga-ahli-section .status-wrapper').length) {
                    $('#skp-tenaga-ahli-section .status-select').hide();
                }
            });
        },

        initModalHandlers: function() {
            console.log('Initializing modal handlers');
            
            // Handle form submission
            $('#status-change-form-tenaga-ahli').on('submit', function(e) {
                e.preventDefault();
                console.log('Status change form submitted');

                if (!AsosiasiSKPUtils.validateForm($(this))) {
                    console.log('Form validation failed');
                    return;
                }

                const formData = new FormData(this);
                formData.append('action', 'update_skp_tenaga_ahli_status');
                formData.append('nonce', $('#skp_tenaga_ahli_nonce').val());

                const $submitBtn = $(this).find('button[type="submit"]');
                const originalText = $submitBtn.text();
                
                console.log('Sending status update request:', {
                    skpId: formData.get('skp_id'),
                    oldStatus: formData.get('old_status'),
                    newStatus: formData.get('new_status')
                });

                $submitBtn.prop('disabled', true)
                         .text(asosiasiSKPTenagaAhli.strings.saving || 'Menyimpan...');

                $.ajax({
                    url: ajaxurl,
                    type: 'POST',
                    data: formData,
                    processData: false,
                    contentType: false,
                    success: function(response) {
                        console.log('Status update response:', response);
                        if (response.success) {
                            AsosiasiSKPUtils.showNotice('success', response.data.message);
                            AsosiasiSKPTenagaAhli.reloadTable(null, 'active');
                            methods.loadStatusHistory();
                        } else {
                            AsosiasiSKPUtils.showNotice('error', response.data.message);
                        }
                        $('#status-change-modal-tenaga-ahli').hide();
                    },
                    error: function(xhr, status, error) {
                        console.error('Status update error:', {xhr, status, error});
                        AsosiasiSKPUtils.showNotice('error', 
                            asosiasiSKPTenagaAhli.strings.statusChangeError || 
                            'Gagal mengubah status SKP'
                        );
                    },
                    complete: function() {
                        $submitBtn.prop('disabled', false).text(originalText);
                    }
                });
            });

            // Modal close handlers
            $('.status-modal-close-tenaga-ahli, .status-modal-cancel-tenaga-ahli').on('click', function() {
                $('#status-change-modal-tenaga-ahli').hide();
            });

            // Close modal on outside click
            $(window).on('click', function(e) {
                if ($(e.target).is('#status-change-modal-tenaga-ahli')) {
                    $('#status-change-modal-tenaga-ahli').hide();
                }
            });
        },

        loadStatusHistory: function() {
            console.log('Loading status history');
            const $historyList = $('#tenaga-ahli-status-history-list');
            const memberId = AsosiasiSKPUtils.getMemberId();

            console.log('History elements:', {
                historyListFound: $historyList.length > 0,
                memberId: memberId,
                nonce: $('#skp_tenaga_ahli_nonce').val()
            });

            if (!$historyList.length || !memberId) {
                console.error('Required elements missing for history load');
                return;
            }

            $historyList.html(`
                <tr class="skp-loading">
                    <td colspan="7" class="text-center">
                        <span class="spinner is-active"></span>
                        <span class="loading-text">
                            ${asosiasiSKPTenagaAhli.strings.loading || 'Memuat riwayat status...'}
                        </span>
                    </td>
                </tr>
            `);

            $.ajax({
                url: ajaxurl,
                type: 'GET',
                data: {
                    action: 'get_skp_tenaga_ahli_status_history',
                    member_id: memberId,
                    nonce: $('#skp_tenaga_ahli_nonce').val()
                },
                success: function(response) {
                    console.log('History response:', response);
                    if (response.success) {
                        methods.renderStatusHistory(response.data.history);
                    } else {
                        AsosiasiSKPUtils.showNotice('error', response.data.message);
                        $historyList.empty();
                    }
                },
                error: function(xhr, status, error) {
                    console.error('History load error:', {xhr, status, error});
                    AsosiasiSKPUtils.showNotice('error', 
                        asosiasiSKPTenagaAhli.strings.loadHistoryError || 
                        'Gagal memuat riwayat status'
                    );
                    $historyList.empty();
                }
            });
        },

        renderStatusHistory: function(history) {
            console.log('Rendering history:', history);
            const $historyList = $('#tenaga-ahli-status-history-list');
            $historyList.empty();

            if (!history || !history.length) {
                $historyList.html(`
                    <tr>
                        <td colspan="7" class="text-center">
                            ${asosiasiSKPTenagaAhli.strings.noHistory || 'Belum ada riwayat perubahan status'}
                        </td>
                    </tr>
                `);
                return;
            }

            history.forEach((item, index) => {
                const oldStatusClass = methods.getStatusClassFromLabel(item.old_status);
                const newStatusClass = methods.getStatusClassFromLabel(item.new_status);
                
                $historyList.append(`
                    <tr>
                        <td>${index + 1}</td>
                        <td>${AsosiasiSKPUtils.escapeHtml(item.nomor_skp)}</td>
                        <td>${AsosiasiSKPUtils.escapeHtml(item.nama_tenaga_ahli)}</td>
                        <td>
                            <span class="skp-status ${oldStatusClass}">
                                ${AsosiasiSKPUtils.escapeHtml(item.old_status)}
                            </span>
                        </td>
                        <td>
                            <span class="skp-status ${newStatusClass}">
                                ${AsosiasiSKPUtils.escapeHtml(item.new_status)}
                            </span>
                        </td>
                        <td>${AsosiasiSKPUtils.escapeHtml(item.reason)}</td>
                        <td>${AsosiasiSKPUtils.escapeHtml(item.changed_by)}</td>
                        <td>${item.changed_at}</td>
                    </tr>
                `);
            });
        },

        getStatusClassFromLabel: function(label) {
            label = label.toLowerCase();
            const statusMap = {
                'aktif': 'skp-status-active',
                'active': 'skp-status-active',
                'diaktifkan': 'skp-status-active', 
                'activated': 'skp-status-active',
                'tidak aktif': 'skp-status-inactive',
                'inactive': 'skp-status-inactive',
                'kadaluarsa': 'skp-status-expired',
                'expired': 'skp-status-expired'
            };
            return statusMap[label] || 'skp-status-inactive';
        }
    };

    // Expose public methods
    AsosiasiSKPTenagaAhliStatus = {
        init: methods.init.bind(methods),
        loadStatusHistory: methods.loadStatusHistory.bind(methods)
    };

    // Initialize when document is ready
    $(document).ready(function() {
        console.log('Document ready in status.js');
        if ($('#skp-tenaga-ahli-section').length) {
            console.log('SKP section found, initializing status handler');
            methods.init();
        } else {
            console.log('SKP section not found');
        }
    });

})(jQuery);
